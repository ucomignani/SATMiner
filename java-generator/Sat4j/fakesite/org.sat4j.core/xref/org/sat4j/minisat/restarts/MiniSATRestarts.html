30  package org.sat4j.minisat.restarts;
31  
32  import org.sat4j.minisat.core.Constr;
33  import org.sat4j.minisat.core.RestartStrategy;
34  import org.sat4j.minisat.core.SearchParams;
35  import org.sat4j.minisat.core.SolverStats;
36  
37  /**
38   * Minisat original restart strategy.
39   * 
40   * @author leberre
41   * 
42   */
43  public final class MiniSATRestarts implements RestartStrategy {
44  
45      /**
46       * 
47       */
48      private static final long serialVersionUID = 1L;
49  
50      private double nofConflicts;
51  
52      private SearchParams params;
53  
54      private int conflictcount;
55  
56      public void init(SearchParams theParams, SolverStats stats) {
57          this.params = theParams;
58          this.nofConflicts = theParams.getInitConflictBound();
59          this.conflictcount = 0;
60      }
61  
62      public long nextRestartNumberOfConflict() {
63          return Math.round(this.nofConflicts);
64      }
65  
66      public void onRestart() {
67          this.nofConflicts *= this.params.getConflictBoundIncFactor();
68      }
69  
70      @Override
71      public String toString() {
72          return "MiniSAT restarts strategy";
73      }
74  
75      public boolean shouldRestart() {
76          return this.conflictcount >= this.nofConflicts;
77      }
78  
79      public void onBackjumpToRootLevel() {
80          this.conflictcount = 0;
81      }
82  
83      public void reset() {
84          this.conflictcount = 0;
85      }
86  
87      public void newConflict() {
88          this.conflictcount++;
89      }
90  
91      public void newLearnedClause(Constr learned, int trailLevel) {
92      }
93  }
