30  package org.sat4j.minisat.orders;
31  
32  import static org.sat4j.core.LiteralsUtils.negLit;
33  
34  import org.sat4j.minisat.core.IPhaseSelectionStrategy;
35  
36  abstract class AbstractPhaserecordingSelectionStrategy implements
37          IPhaseSelectionStrategy {
38  
39      /**
40  	 * 
41  	 */
42      private static final long serialVersionUID = 1L;
43  
44      protected int[] phase;
45  
46      public void init(int nlength) {
47          if (this.phase == null || this.phase.length < nlength) {
48              this.phase = new int[nlength];
49          }
50          for (int i = 1; i < nlength; i++) {
51              this.phase[i] = negLit(i);
52          }
53      }
54  
55      public void init(int var, int p) {
56          this.phase[var] = p;
57      }
58  
59      public int select(int var) {
60          return this.phase[var];
61      }
62  }
