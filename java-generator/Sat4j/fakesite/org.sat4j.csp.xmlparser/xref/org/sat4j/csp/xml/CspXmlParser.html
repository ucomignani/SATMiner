22  package org.sat4j.csp.xml;
23  
24  import java.io.FileReader;
25  import java.io.IOException;
26  import java.net.URL;
27  
28  import javax.xml.parsers.ParserConfigurationException;
29  import javax.xml.parsers.SAXParserFactory;
30  import javax.xml.validation.SchemaFactory;
31  
32  import org.xml.sax.InputSource;
33  import org.xml.sax.SAXException;
34  import org.xml.sax.XMLReader;
35  import org.xml.sax.helpers.DefaultHandler;
36  
37  public class CspXmlParser {
38  
39  	public static void parse(ICSPCallback callback, String filename)
40  			throws SAXException, ParserConfigurationException, IOException {
41  		// le producteur
42  		XMLReader prod;
43  		// le consommateur
44  		DefaultHandler cons;
45  
46  		// obtenir un parser
47  		SAXParserFactory saxpf = SAXParserFactory.newInstance();
48  		// associer un schema pour la validation
49  		URL url = CspXmlParser.class.getResource("/instance_2_0.xsd");
50  		// System.out.println(url);
51  		saxpf.setSchema(SchemaFactory.newInstance(
52  				javax.xml.XMLConstants.W3C_XML_SCHEMA_NS_URI).newSchema(url));
53  		prod = saxpf.newSAXParser().getXMLReader();
54  
55  		cons = new InstanceParser(callback);
56  		prod.setContentHandler(cons);
57  		prod.setErrorHandler(cons);
58  
59  		// et c'est parti!
60  		prod.parse(new InputSource(new FileReader(filename)));
61  	}
62  }
