22  package org.sat4j.csp.xml;
23  import org.xml.sax.Attributes;
24  
25  import static org.sat4j.csp.xml.TagNames.*;
26  
27  class Constraint extends Element {
28  
29      public Constraint(ICSPCallback out,String tagName) {
30  		super(out,tagName);
31  	}
32  
33  	public void startElement(Attributes att){
34  		getCB().beginConstraint(att.getValue(NAME), Integer.parseInt(att.getValue(ARITY)));
35  		String ref = att.getValue(REFERENCE);
36  		if (ref != null)
37  			getCB().constraintReference(ref);
38  		String scope = att.getValue(SCOPE);
39  		if (scope != null){
40  			// traitement des variables dans le scope
41  			String[] tokens = scope.trim().split("\\s+");
42  			for(String tok : tokens)
43  				if (!tok.equals(""))
44  					getCB().addVariableToConstraint(tok);	
45  		}
46  		context = "Constraint";
47  	}
48  
49  	public void endElement() {
50  		getCB().endConstraint();
51  		context = "";
52  	}
53  
54  	private String context;
55  	
56  	public String getContexte(){
57  		return context;
58  	}
